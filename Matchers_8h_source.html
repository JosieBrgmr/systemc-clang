183 void processInstanceCXXDecls(ASTContext &context) {
184 // Must have found instances.
185 // 1. For every module found, check if there is an instance.
186 // 2. If there is an instance, then add it into the list.
187
188 llvm::outs() << "###### DUMP Instance Matches \n";
189 instance_matcher_.dump();
190
191 auto instance_map{instance_matcher_.getInstanceMap()};
192
198 matchInstancesInBaseClasses(context);
199
201 instance_map = instance_matcher_.getInstanceMap();
202
204 for (const auto &inst: instance_map) {
205 llvm::dbgs()
206 << "@@@@@@@@@@@@@ INSTANCE MAP with base instances @@@@@@@@@@@\n";
207
208 clang::CXXRecordDecl *decl{
209 dyn_cast<clang::CXXRecordDecl>(inst.second.getInstanceTypeDecl())};
210 auto name{decl->getNameAsString()};
211 llvm::dbgs() << "class: " << name << "\n";
212 }
214
217 for (const auto &inst: instance_map) {
218 ModuleInstanceType instance{inst.second};
219
220 clang::CXXRecordDecl *decl{
221 dyn_cast<clang::CXXRecordDecl>(inst.second.getInstanceTypeDecl())};
222 auto name{decl->getNameAsString()};
223 llvm::outs() << "############### ====> INST: " << inst.first
224 << ", name: " << name
225 << ", instance_name: " << inst.second.instance_name << "\n";
226
227 auto add_module{new ModuleInstance(name, decl)};
228 add_module->setInstanceInfo(instance);

231 modules_.insert(std::pair<clang::CXXRecordDecl *, ModuleInstance *>(
232 decl, add_module));
233
234

236 runModuleDeclarationMatchers(context, decl, add_module);
237 llvm::dbgs() << "[Running port matcher]\n";
238 runPortMatcher(context, decl, add_module);
239
240 llvm::dbgs() << "[Running Base class logic]\n";
244 auto base_decls{getAllBaseClasses(decl)};

246 auto name{base_decl->getNameAsString()};
