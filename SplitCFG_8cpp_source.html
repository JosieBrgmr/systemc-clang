29void SplitCFG::dfs_visit_wait(
30 const SplitCFGBlock* BB,
31 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_blocks,
32 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_to_visit,
33 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits,
34 llvm::SmallVector<std::pair<const SplitCFGBlock*, SplitCFGPathInfo>>&
35 curr_path) {
37 if (BB->succ_empty()) {
38 return;
39 }
40
41 popping_ = false;
42 // successors to visit
43 llvm::SmallVector<
44 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
45 to_visit;
46
47 visited_blocks.insert(BB);
48 to_visit.push_back(std::make_pair(BB, BB->succ_begin()));
49
50 do {
51 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
52 to_visit.back();
53 const SplitCFGBlock* ParentBB{Top.first};
54 SplitCFGBlock::const_succ_iterator& I{Top.second};
55
57 bool bb_has_wait{(ParentBB->hasWait())};
58
60 if (!popping_) {
61 llvm::dbgs() << "Visit Parent BB# " << ParentBB->getBlockID() << " ";
62 curr_path.push_back(std::make_pair(ParentBB, SplitCFGPathInfo{ParentBB}));
63 } else {
64 }
65
68 if (bb_has_wait) {
69 llvm::dbgs() << " has WAIT ";
70 bool found_suc{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
71 // A wait to be visited is added.
72 if (visited_waits.insert(ParentBB).second) {
73 waits_to_visit.insert(waits_to_visit.begin(), BB);
74 ++next_state_count_;
75 auto wit{wait_next_state_.find(ParentBB)};
76 if (wit == wait_next_state_.end()) {
77 wait_next_state_.insert(
78 std::make_pair(ParentBB, std::make_pair(BB, next_state_count_)));
79 }
80 }
81 }
82
83 // Current block is a loop and has two successors. So, we should start with
84 // a new call to dfs, and provide a new visited_blocks.
85 //
86 // If there is a successor that has not been visited, then remember that
87 // block.
88 bool found_succ{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
89 // llvm::dbgs() << "Found successor BB " << BB->getBlockID()
90 // << " for parentBB " << ParentBB->getBlockID() << "\n";
91
92 // FIXME: Should this have found_succ &&?
93 if (isLoopWithTwoSuccessors(ParentBB)) {
94 do {
95 // llvm::dbgs() << "\n==============================================";
96 llvm::dbgs() << "\n#### BB " << ParentBB->getBlockID()
97 << " is a loop with 2 succ\n";
98 llvm::SmallPtrSet<const SplitCFGBlock*, 32> loop_visited_blocks;
99 // ParentBB has been visited so don't revisit it
100 loop_visited_blocks.insert(ParentBB);
101 // dumpVisitedBlocks(visited_blocks);
102 visited_blocks.insert(BB);
103 // llvm::dbgs() << "\n==============================================";
104 // llvm::dbgs() << "\nRecurse DFS starting at BB " << BB->getBlockID()
105 // << " visited_block size " << visited_blocks.size() <<
106 // "\n";
107 dfs_visit_wait(BB, loop_visited_blocks, waits_to_visit, visited_waits,
108 curr_path);
109
112 updateVisitedBlocks(loop_visited_blocks, loop_visited_blocks);
113 // dumpVisitedBlocks(visited_blocks);
114 // llvm::dbgs() << "\nEND Recurse DFS"
115 // << " visited_block size " << visited_blocks.size() <<
116 // "\n";
117 // llvm::dbgs() << "\n==============================================";
118
127 //
128
129 found_succ = getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB);
130 // llvm::dbgs() << "unvisited BB " << BB->getBlockID() << " is succ "
131 // << found_succ << "\n";
132 //
135 if (!found_succ) {
136 updateVisitedBlocks(visited_blocks, loop_visited_blocks);
137 }
138 } while (found_succ);
139
140 } else {
141 // Recursive call in the if-then will traverse the subgraph.
142 // Only insert successor if recursive call does not visit subgraph.
143 }
144 addSuccessorToVisitOrPop(bb_has_wait, BB, to_visit, found_succ);
145 // std::cin.get();
146 // llvm::dbgs() << "to_visit ";
147 // dumpSmallVector(to_visit);
148 // llvm::dbgs() << " End loop \n";
149 } while (!to_visit.empty());
150}

169bool SplitCFG::isLoopWithTwoSuccessors(const SplitCFGBlock* block) const {
170 // The number of successor has to be exactly 2 in order to create a new
171 // visited_blocks.
172 //
173 // Note that CFGBlock often times has a NULL successor. We have to ignore
174 // that.
175 auto cfg_block{block->getCFGBlock()};
176 bool last_succ_is_null{false};
177 if (cfg_block->succ_size() == 2) {
178 // Check that the last one is not NULL
179
180 if (*cfg_block->succ_rbegin() == nullptr) {
181 last_succ_is_null = true;
182 }
183 }
184
185 return (block && isLoop(block) && (last_succ_is_null == false));
186}

308void SplitCFG::dfs_pop_on_wait(
309 const SplitCFGBlock* basic_block,
310 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_in_stack,
311 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits) {
313 if (basic_block->succ_empty()) {
314 return;
315 }
316
317 bool walking_up_from_wait{false};
318 llvm::SmallPtrSet<const SplitCFGBlock*, 8> visited;
319 llvm::SmallVector<
320 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
321 visit_stack;
322 llvm::SmallVector<const SplitCFGBlock*, 8> in_stack;
323
324 visited.insert(basic_block);
325 visit_stack.push_back(std::make_pair(basic_block, basic_block->succ_begin()));
326
327 // llvm::SmallVector<const SplitCFGBlock*> curr_path;
328 VectorSplitCFGBlock curr_path;
329 do {
330 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
331 visit_stack.back();
332 const SplitCFGBlock* parent_bb = Top.first;
333 SplitCFGBlock::const_succ_iterator& I = Top.second;
334
336 bool bb_has_wait{(parent_bb->hasWait())};
337
338 if (bb_has_wait) {
339 walking_up_from_wait = bb_has_wait;
340 llvm::dbgs() << "BB# " << parent_bb->getBlockID()
341 << " has a wait in it\n";
342 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
343 curr_path.push_back(parent_bb);
344
347 if (visited_waits.insert(parent_bb).second == true) {
348 // Insert the successor of the block.
349
350 while (I != parent_bb->succ_end()) {
351 basic_block = *I++;
352 if ((basic_block != nullptr)) {
353 llvm::dbgs() << "Insert successor of BB#" << parent_bb->getBlockID()
354 << ": BB#" << basic_block->getBlockID() << "\n";
355 // Mimic FIFO
356 waits_in_stack.insert(waits_in_stack.begin(), basic_block);
357 auto wit = wait_next_state_.find(parent_bb);
358 if (wit == wait_next_state_.end()) {
359 wait_next_state_.insert(std::make_pair(
360 parent_bb, std::make_pair(basic_block, next_state_count_)));
361 ++next_state_count_;
362 }
363 break;
364 }
365 }
366 }
367 }
368
369 // A new block is found only if the block is not visited, and it is not a
370 // wait(). If yes, then mark it as visited.
371 bool FoundNew = false;
372 while (I != parent_bb->succ_end()) {
373 basic_block = *I++;
374 if ((basic_block != nullptr) && (!bb_has_wait) &&
375 (visited.insert(basic_block).second)) {
376 FoundNew = true;
377 break;
378 }
379 }
380
381 if (FoundNew) {
382 // If a new successor has been found, then go down one level if there is
383 // a unvisited successor.
384 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
385 visit_stack.push_back(
386 std::make_pair(basic_block, basic_block->succ_begin()));
387 curr_path.push_back(parent_bb);
388 if (walking_up_from_wait) {
389 visited.clear();
390 // Reinsert the successor block
391 visited.insert(basic_block);
392 walking_up_from_wait = false;
393 }
394 } else {
395 // Go up one level.
396 // llvm::dbgs() << "pop: " << visit_stack.size() << "\n";
397 visit_stack.pop_back();
398 }
399 } while (!visit_stack.empty());
400
402 paths_found_.push_back(curr_path);
403}

406 if (auto cfg_stmt = element.getAs<clang::CFGStmt>()) {
407 auto stmt{cfg_stmt->getStmt()};
408
409 if (auto* expr = llvm::dyn_cast<clang::Expr>(stmt)) {

419 return IntLiteral->getValue();
420 }
421 }

458void SplitCFG::splitBlock(clang::CFGBlock* block) {
459 assert(block != nullptr);
460
461 unsigned int num_elements{block->size()};
462
463 llvm::dbgs() << "\n\nChecking if block " << block->getBlockID()
464 << " has a wait()\n";
465 llvm::dbgs() << "Number of elements " << num_elements << "\n";
466 // We are going to have two vectors.
467 // 1. A vector of vector pointers to CFGElements.
468 // 2. A vector of pointers to CFGElements that are waits.
469 //
470
471 VectorCFGElementPtr vec_elements{};
472 llvm::SmallVector<std::pair<VectorCFGElementPtr, bool>> split_elements;
473
475 if (num_elements == 0) {
476 split_elements.push_back(std::make_pair(vec_elements, false));
477 } else {
478 bool has_wait{false};
479 for (auto const& element : block->refs()) {
480 has_wait = false;
481 llvm::dbgs() << "Is this a wait? ";
482 element->dump();
483
487 const clang::CFGElement* element_ptr{element.operator->()};
489 if (isElementWait(*element)) {
490 llvm::dbgs() << " IT IS A WAIT\n";
492 if (num_elements == 1) {
493 llvm::dbgs() << "DBG: Only one statement and it is a wait().\n";
494 }
495
496 // Add all elements before the wait into the split elements.
497 if (vec_elements.size() != 0) {
498 split_elements.push_back(std::make_pair(vec_elements, false));
499 vec_elements.clear();
500 }
501
503 vec_elements.push_back(element_ptr);
504 split_elements.push_back(std::make_pair(vec_elements, true));
505 vec_elements.clear();
506
507 has_wait = true;
508 } else {
509 vec_elements.push_back(element_ptr);
510 }
511 }
512
513 if (vec_elements.size() != 0) {
514 split_elements.push_back(std::make_pair(vec_elements, has_wait));
515 }
516 }
517
518 createWaitSplitCFGBlocks(block, split_elements);
519}

521void SplitCFG::createWaitSplitCFGBlocks(
522 clang::CFGBlock* block,
523 const llvm::SmallVectorImpl<std::pair<SplitCFG::VectorCFGElementPtr, bool>>&
524 split_elements) {
525 // dumpSplitElements(split_elements);
527 llvm::dbgs() << "Number of entries in split_elements "
528 << split_elements.size() << "\n";
529 unsigned int id{0};
530 SplitCFGBlock* prev_block{nullptr};
531
532 for (auto const& elements : split_elements) {
533 llvm::dbgs() << "Element number " << id << " ";
535 SplitCFGBlock* new_split{nullptr};
536
537 bool already_exists{false};
538 // id = 0 is when it is the first sequence of elements.
539 if (id == 0) {
540 auto scit{sccfg_.find(block->getBlockID())};
541 // Must be true.
542 // bool already_exists{scit != sccfg_.end()};
543 new_split = scit->second;
544 new_split->id_ = block->getBlockID();
545
546 // Successor: succ block from CFGBlock (overridden)
548 if (split_elements.size() == 1) {
549 addSuccessors(new_split, block);
550 }
551 // Prececessor: prev block from CFGBlock
552 addPredecessors(new_split, block);
553 } else {
554 new_split = new SplitCFGBlock{};
555 new_split->id_ = block->getBlockID() * 10 + id;
556
558 //
560 llvm::dbgs() << "SB# " << new_split->id_ << " size "

562 if (id == split_elements.size() - 1) {
563 llvm::dbgs() << " add successors of BB# " << block->getBlockID()
564 << "\n";
565 addSuccessors(new_split, block);
566 }
567 prev_block->successors_.push_back(new_split);
569 new_split->predecessors_.push_back(prev_block);
570 sccfg_.insert(std::make_pair(block->getBlockID() * 10 + id, new_split));
571 }
572
573 new_split->block_ = block;
574 new_split->has_wait_ = elements.second;
575 new_split->elements_ = elements.first;
576
577 // If it's a wait() then get the argument.
578 if (new_split->has_wait_) {
580 const clang::CFGElement* wait_el = (*new_split->elements_.begin());
581 new_split->wait_arg_ = getWaitArgument(*wait_el);
582 }
583
584 // Set the successor of new_split.
585 prev_block = new_split;
586 ++id;
587 }

734 cfg_->dump(lang_opts, true);

766void SplitCFG::dumpToDot() const {

768 std::error_code ok;

770 llvm::sys::fs::OF_None);
771
772 dotos << "digraph SCCFG {\n";
773 dotos << " rankdir=TD\n";
774 dotos << " node [shape=record]\n";
776 for (auto const& block : sccfg_) {
777 const SplitCFGBlock* sblock{block.second};
779
780 std::string element_str{};
781 llvm::raw_string_ostream element_os(element_str);
782 auto num_elements{sblock->getElements().size()};
783 auto i{0};
784
785 element_os << "|{";
786 for (auto const& element : sblock->getElements()) {
787 element_os << "| " << i << ":";
788 element->dumpToStream(element_os);
789 ++i;
790 // if (i < num_elements) {
791 // element_os << "|";
792 // }
793 }
794 element_os << "}";
795
796 std::regex re("\\-");
797 element_str = std::regex_replace(element_str, re, "\\-");
798 std::regex replus("\\+");
799 element_str = std::regex_replace(element_str, replus, "\\+");
800 std::regex relt("\\<");
801 element_str = std::regex_replace(element_str, relt, "\\<");
802 std::regex regt("\\>");
803 element_str = std::regex_replace(element_str, regt, "\\>");
804 std::regex reamp("\\&");
805 element_str = std::regex_replace(element_str, reamp, "\\&");
806
807 if (sblock->hasWait()) {
808 dotos << "SB" << sblock->getBlockID() << " [ \n color=red, label=\"SB"
809 << sblock->getBlockID() << "\n"
810 << " [" << sblock->getNextState() << "] | WAIT " << element_str
811 << "\"\n]"
812 << "\n";
813 } else {
814 if (isLoop(sblock)) {
815 auto terminator{sblock->getCFGBlock()->getTerminatorStmt()};
816 if (llvm::isa<clang::WhileStmt>(terminator)) {
817 element_str += " | WHILE ";
818 }
819 if (llvm::isa<clang::ForStmt>(terminator)) {
820 element_str += " | FOR ";
821 }
822 if (llvm::isa<clang::DoStmt>(terminator)) {
823 element_str += " | DOWHILE ";
824 }
825 // sblock->getCFGBlock()->getTerminatorStmt()->dumpPretty(context);
826 // auto stmt{sblock->getCFGBlock()->getTerminatorStmt()};
827 // stmt->printPretty(dotos, nullptr,
828 // clang::PrintingPolicy(context_.getLangOpts()));
829 }
830
831 if (isConditional(sblock)) {
832 auto terminator{sblock->getCFGBlock()->getTerminatorStmt()};
833 if (llvm::isa<clang::IfStmt>(terminator)) {
834 element_str += " | IF ";
835 }
836 }
837
838 dotos << "SB" << sblock->getBlockID() << " [ \n label=\"SB"
839 << sblock->getBlockID() << "\n"
840 << element_str << "\"\n]"
841 << "\n";
842 }
843 }
845 for (auto const& block : sccfg_) {
846 SplitCFGBlock* sblock{block.second};
847 for (auto const& succ : sblock->successors_) {
848 dotos << "SB" << sblock->getBlockID();
849 dotos << " -> SB" << succ->getBlockID() << "\n";
850 }
851 }
852 dotos << "}\n";
853}
