29void SplitCFG::dfs_visit_wait(
30 const SplitCFGBlock* BB,
31 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_blocks,
32 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_to_visit,
33 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits,
34 llvm::SmallVector<std::pair<const SplitCFGBlock*, SplitCFGPathInfo>>&
35 curr_path) {
37 if (BB->succ_empty()) {
38 return;
39 }
40
41 popping_ = false;
42 // successors to visit
43 llvm::SmallVector<
44 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
45 to_visit;
46
47 visited_blocks.insert(BB);
48 to_visit.push_back(std::make_pair(BB, BB->succ_begin()));
49
50 do {
51 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
52 to_visit.back();
53 const SplitCFGBlock* ParentBB{Top.first};
54 SplitCFGBlock::const_succ_iterator& I{Top.second};
55
57 bool bb_has_wait{(ParentBB->hasWait())};
58
60 if (!popping_) {
61 llvm::dbgs() << "Visit Parent BB# " << ParentBB->getBlockID() << " ";
62 curr_path.push_back(std::make_pair(ParentBB, SplitCFGPathInfo{ParentBB}));
63 } else {
64 }
65
68 if (bb_has_wait) {
69 llvm::dbgs() << " has WAIT ";
70 bool found_suc{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
71 // A wait to be visited is added.
72 if (visited_waits.insert(ParentBB).second) {
73 waits_to_visit.insert(waits_to_visit.begin(), BB);
74 }
75 }
76
77 // Current block is a loop and has two successors. So, we should start with
78 // a new call to dfs, and provide a new visited_blocks.
79 //
80 // If there is a successor that has not been visited, then remember that
81 // block.
82 bool found_succ{getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB)};
83 // llvm::dbgs() << "Found successor BB " << BB->getBlockID()
84 // << " for parentBB " << ParentBB->getBlockID() << "\n";
85
86 // FIXME: Should this have found_succ &&?
87 if (isLoopWithTwoSuccessors(ParentBB)) {
88 do {
89 // llvm::dbgs() << "\n==============================================";
90 llvm::dbgs() << "\n#### BB " << ParentBB->getBlockID()
91 << " is a loop with 2 succ\n";
92 llvm::SmallPtrSet<const SplitCFGBlock*, 32> loop_visited_blocks;
93 // ParentBB has been visited so don't revisit it
94 loop_visited_blocks.insert(ParentBB);
95 // dumpVisitedBlocks(visited_blocks);
96 visited_blocks.insert(BB);
97 // llvm::dbgs() << "\n==============================================";
98 // llvm::dbgs() << "\nRecurse DFS starting at BB " << BB->getBlockID()
99 // << " visited_block size " << visited_blocks.size() <<
100 // "\n";
101 dfs_visit_wait(BB, loop_visited_blocks, waits_to_visit, visited_waits,
102 curr_path);
103
106 updateVisitedBlocks(loop_visited_blocks, loop_visited_blocks);
107 // dumpVisitedBlocks(visited_blocks);
108 // llvm::dbgs() << "\nEND Recurse DFS"
109 // << " visited_block size " << visited_blocks.size() <<
110 // "\n";
111 // llvm::dbgs() << "\n==============================================";
112
121 //
122
123 found_succ = getUnvisitedSuccessor(ParentBB, I, visited_blocks, BB);
124 // llvm::dbgs() << "unvisited BB " << BB->getBlockID() << " is succ "
125 // << found_succ << "\n";
126 //
129 if (!found_succ) {
130 updateVisitedBlocks(visited_blocks, loop_visited_blocks);
131 }
132 } while (found_succ);
133
134 } else {
135 // Recursive call in the if-then will traverse the subgraph.
136 // Only insert successor if recursive call does not visit subgraph.
137 }
138 addSuccessorToVisitOrPop(bb_has_wait, BB, to_visit, found_succ);
139 // std::cin.get();
140 // llvm::dbgs() << "to_visit ";
141 // dumpSmallVector(to_visit);
142 // llvm::dbgs() << " End loop \n";
143 } while (!to_visit.empty());
144}

163bool SplitCFG::isLoopWithTwoSuccessors(const SplitCFGBlock* block) const {
164 // The number of successor has to be exactly 2 in order to create a new
165 // visited_blocks.
166 //
167 // Note that CFGBlock often times has a NULL successor. We have to ignore
168 // that.
169 auto cfg_block{block->getCFGBlock()};
170 bool last_succ_is_null{false};
171 if (cfg_block->succ_size() == 2) {
172 // Check that the last one is not NULL
173
174 if (*cfg_block->succ_rbegin() == nullptr) {
175 last_succ_is_null = true;
176 }
177 }
178
179 return (block && isLoop(block) && (last_succ_is_null == false));
180}

293void SplitCFG::dfs_pop_on_wait(
294 const SplitCFGBlock* basic_block,
295 llvm::SmallVectorImpl<const SplitCFGBlock*>& waits_in_stack,
296 llvm::SmallPtrSetImpl<const SplitCFGBlock*>& visited_waits) {
298 if (basic_block->succ_empty()) {
299 return;
300 }
301
302 bool walking_up_from_wait{false};
303 llvm::SmallPtrSet<const SplitCFGBlock*, 8> visited;
304 llvm::SmallVector<
305 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>, 8>
306 visit_stack;
307 llvm::SmallVector<const SplitCFGBlock*, 8> in_stack;
308
309 visited.insert(basic_block);
310 visit_stack.push_back(std::make_pair(basic_block, basic_block->succ_begin()));
311
312 // llvm::SmallVector<const SplitCFGBlock*> curr_path;
313 VectorSplitCFGBlock curr_path;
314 do {
315 std::pair<const SplitCFGBlock*, SplitCFGBlock::const_succ_iterator>& Top =
316 visit_stack.back();
317 const SplitCFGBlock* parent_bb = Top.first;
318 SplitCFGBlock::const_succ_iterator& I = Top.second;
319
321 bool bb_has_wait{(parent_bb->hasWait())};
322
323 if (bb_has_wait) {
324 walking_up_from_wait = bb_has_wait;
325 llvm::dbgs() << "BB# " << parent_bb->getBlockID()
326 << " has a wait in it\n";
327 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
328 curr_path.push_back(parent_bb);
329
332 if (visited_waits.insert(parent_bb).second == true) {
333 // Insert the successor of the block.
334
335 while (I != parent_bb->succ_end()) {
336 basic_block = *I++;
337 if ((basic_block != nullptr)) {
338 llvm::dbgs() << "Insert successor of BB#" << parent_bb->getBlockID()
339 << ": BB#" << basic_block->getBlockID() << "\n";
340 // Mimic FIFO
341 waits_in_stack.insert(waits_in_stack.begin(), basic_block);
342 auto wit = wait_next_state_.find(parent_bb);
343 if (wit == wait_next_state_.end()) {
344 wait_next_state_.insert(std::make_pair(
345 parent_bb, std::make_pair(basic_block, next_state_count_)));
346 ++next_state_count_;
347 }
348 break;
349 }
350 }
351 }
352 }
353
354 // A new block is found only if the block is not visited, and it is not a
355 // wait(). If yes, then mark it as visited.
356 bool FoundNew = false;
357 while (I != parent_bb->succ_end()) {
358 basic_block = *I++;
359 if ((basic_block != nullptr) && (!bb_has_wait) &&
360 (visited.insert(basic_block).second)) {
361 FoundNew = true;
362 break;
363 }
364 }
365
366 if (FoundNew) {
367 // If a new successor has been found, then go down one level if there is
368 // a unvisited successor.
369 llvm::dbgs() << "Visited BB#" << parent_bb->getBlockID() << "\n";
370 visit_stack.push_back(
371 std::make_pair(basic_block, basic_block->succ_begin()));
372 curr_path.push_back(parent_bb);
373 if (walking_up_from_wait) {
374 visited.clear();
375 // Reinsert the successor block
376 visited.insert(basic_block);
377 walking_up_from_wait = false;
378 }
379 } else {
380 // Go up one level.
381 // llvm::dbgs() << "pop: " << visit_stack.size() << "\n";
382 visit_stack.pop_back();
383 }
384 } while (!visit_stack.empty());
385
387 paths_found_.push_back(curr_path);
388}

443void SplitCFG::splitBlock(clang::CFGBlock* block) {
444 assert(block != nullptr);
445
446 unsigned int num_elements{block->size()};
447
448 llvm::dbgs() << "\n\nChecking if block " << block->getBlockID()
449 << " has a wait()\n";
450 llvm::dbgs() << "Number of elements " << num_elements << "\n";
451 // We are going to have two vectors.
452 // 1. A vector of vector pointers to CFGElements.
453 // 2. A vector of pointers to CFGElements that are waits.
454 //
455
456 VectorCFGElementPtr vec_elements{};
457 llvm::SmallVector<std::pair<VectorCFGElementPtr, bool>> split_elements;
458
460 if (num_elements == 0) {
461 split_elements.push_back(std::make_pair(vec_elements, false));
462 } else {
463 bool has_wait{false};
464 for (auto const& element : block->refs()) {
465 has_wait = false;
466 llvm::dbgs() << "Is this a wait? ";
467 element->dump();
468
472 const clang::CFGElement* element_ptr{element.operator->()};
474 if (isElementWait(*element)) {
475 llvm::dbgs() << " IT IS A WAIT\n";
477 if (num_elements == 1) {
478 llvm::dbgs() << "DBG: Only one statement and it is a wait().\n";
479 }
480
481 // Add all elements before the wait into the split elements.
482 if (vec_elements.size() != 0) {
483 split_elements.push_back(std::make_pair(vec_elements, false));
484 vec_elements.clear();
485 }
486
488 vec_elements.push_back(element_ptr);
489 split_elements.push_back(std::make_pair(vec_elements, true));
490 vec_elements.clear();
491
492 has_wait = true;
493 } else {
494 vec_elements.push_back(element_ptr);
495 }
496 }
497
498 if (vec_elements.size() != 0) {
499 split_elements.push_back(std::make_pair(vec_elements, has_wait));
500 }
501 }
502
503 createWaitSplitCFGBlocks(block, split_elements);
504}

506void SplitCFG::createWaitSplitCFGBlocks(
507 clang::CFGBlock* block,
508 const llvm::SmallVectorImpl<std::pair<SplitCFG::VectorCFGElementPtr, bool>>&
509 split_elements) {
510 // dumpSplitElements(split_elements);
512 llvm::dbgs() << "Number of entries in split_elements "
513 << split_elements.size() << "\n";
514 unsigned int id{0};
515 SplitCFGBlock* prev_block{nullptr};
516
517 for (auto const& elements : split_elements) {
518 llvm::dbgs() << "Element number " << id << " ";
520 SplitCFGBlock* new_split{nullptr};
521
522 bool already_exists{false};
523 // id = 0 is when it is the first sequence of elements.
524 if (id == 0) {
525 auto scit{sccfg_.find(block->getBlockID())};
526 // Must be true.
527 // bool already_exists{scit != sccfg_.end()};
528 new_split = scit->second;
529 new_split->id_ = block->getBlockID();
530
531 // Successor: succ block from CFGBlock (overridden)
533 if (split_elements.size() == 1) {
534 addSuccessors(new_split, block);
535 }
536 // Prececessor: prev block from CFGBlock
537 addPredecessors(new_split, block);
538 } else {
539 new_split = new SplitCFGBlock{};
540 new_split->id_ = block->getBlockID() * 10 + id;
541
543 //
545 llvm::dbgs() << "SB# " << new_split->id_ << " size "
546 << split_elements.size() << "\n";
547 if (id == split_elements.size() - 1) {
548 llvm::dbgs() << " add successors of BB# " << block->getBlockID()
549 << "\n";
550 addSuccessors(new_split, block);
551 }
552 prev_block->successors_.push_back(new_split);
554 new_split->predecessors_.push_back(prev_block);
555 sccfg_.insert(std::make_pair(block->getBlockID() * 10 + id, new_split));
556 }
557
558 new_split->block_ = block;
559 new_split->has_wait_ = elements.second;
560 new_split->elements_ = elements.first;
561
562 // If it's a wait() then get the argument.
563 if (new_split->has_wait_) {
565 const clang::CFGElement* wait_el = (*new_split->elements_.begin());
566 new_split->wait_arg_ = getWaitArgument(*wait_el);
567 }
568
569 // Set the successor of new_split.
570 prev_block = new_split;
571 ++id;
572 }

745void SplitCFG::dumpToDot() const {
746 std::error_code OutErrorInfo;
747 std::error_code ok;
748 llvm::raw_fd_ostream dotos(llvm::StringRef("sccfg.dot"), OutErrorInfo,
749 llvm::sys::fs::OF_None);
750
751 dotos << "digraph SCCFG {\n";
752 dotos << " rankdir=TD\n";
753 dotos << " node [shape=record]\n";
755 for (auto const& block : sccfg_) {
756 const SplitCFGBlock* sblock{block.second};
758
759 std::string element_str{};
760 llvm::raw_string_ostream element_os(element_str);
761 auto num_elements{sblock->getElements().size()};
762 auto i{0};
763
764 element_os << "|{";
765 for (auto const& element : sblock->getElements()) {
766 element_os << "| " << i << ":";
767 element->dumpToStream(element_os);
768 ++i;
769 // if (i < num_elements) {
770 // element_os << "|";
771 // }
772 }
773 element_os << "}";
774
775 std::regex re("\\-");
776 element_str = std::regex_replace(element_str, re, "\\-");
777 std::regex replus("\\+");
778 element_str = std::regex_replace(element_str, replus, "\\+");
779 std::regex relt("\\<");
780 element_str = std::regex_replace(element_str, relt, "\\<");
781 std::regex regt("\\>");
782 element_str = std::regex_replace(element_str, regt, "\\>");
783 std::regex reamp("\\&");
784 element_str = std::regex_replace(element_str, reamp, "\\&");
785
786 if (sblock->hasWait()) {
787 dotos << "SB" << sblock->getBlockID() << " [ \n color=red, label=\"SB"
788 << sblock->getBlockID() << "\n"
789 << " [" << sblock->getNextState() << "] | WAIT " << element_str
790 << "\"\n]"
791 << "\n";
792 } else {
793 if (isLoop(sblock)) {
794 auto terminator{sblock->getCFGBlock()->getTerminatorStmt()};
795 if (llvm::isa<clang::WhileStmt>(terminator)) {
796 element_str += " | WHILE ";
797 }
798 if (llvm::isa<clang::ForStmt>(terminator)) {
799 element_str += " | FOR ";
800 }
801 if (llvm::isa<clang::DoStmt>(terminator)) {
802 element_str += " | DOWHILE ";
803 }
804 // sblock->getCFGBlock()->getTerminatorStmt()->dumpPretty(context);
805 // auto stmt{sblock->getCFGBlock()->getTerminatorStmt()};
806 // stmt->printPretty(dotos, nullptr,
807 // clang::PrintingPolicy(context_.getLangOpts()));
808 }
809
810 if (isConditional(sblock)) {
811 auto terminator{sblock->getCFGBlock()->getTerminatorStmt()};
812 if (llvm::isa<clang::IfStmt>(terminator)) {
813 element_str += " | IF ";
814 }
815 }
816
817 dotos << "SB" << sblock->getBlockID() << " [ \n label=\"SB"
818 << sblock->getBlockID() << "\n"
819 << element_str << "\"\n]"
820 << "\n";
821 }
822 }
824 for (auto const& block : sccfg_) {
825 SplitCFGBlock* sblock{block.second};
826 for (auto const& succ : sblock->successors_) {
827 dotos << "SB" << sblock->getBlockID();
828 dotos << " -> SB" << succ->getBlockID() << "\n";
829 }
830 }
831 dotos << "}\n";
832}
