33class BreakMatcher : public MatchFinder::MatchCallback {
34 private:
35 bool break_found_;
36 bool wait_found_;
37
38 public:
39 bool hasBreak() const { return break_found_; }
40 bool hasWait() const { return wait_found_; }
41
42 void registerMatchers(MatchFinder &finder) {
43 // clang-format off
44 //auto bstmt = stmt(hasDescendant(breakStmt().bind("break_stmt")));
45 auto bstmt = findAll(breakStmt().bind("break_stmt"));
46 auto wait_stmt = findAll(cxxMemberCallExpr(callee(cxxMethodDecl(hasName("wait")))).bind("wait_stmt"));
47 // clang-format on
48
49 finder.addMatcher(bstmt, this);
50 finder.addMatcher(wait_stmt, this);
51 }
52
53 virtual void run(const MatchFinder::MatchResult &result) {
54 auto bk_stmt = const_cast<clang::Stmt *>(
55 result.Nodes.getNodeAs<clang::Stmt>("break_stmt"));
56 auto wait_stmt = const_cast<clang::CXXMemberCallExpr *>(
57 result.Nodes.getNodeAs<clang::CXXMemberCallExpr>("wait_stmt"));
58
59 if (bk_stmt) {
60 LLVM_DEBUG(llvm::dbgs() << "## BreakStmt \n");
61 break_found_ = true;
62 }
63 if (wait_stmt) {
64 LLVM_DEBUG(llvm::dbgs() << "## Wait\n");
65 wait_found_ = true;
66 }
67 }
68
69 void dump() {
70 if (break_found_) {
71 LLVM_DEBUG(llvm::dbgs() << "## BREAK FOUND\n");
72 }
73 }
74};
